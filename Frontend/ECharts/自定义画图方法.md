#### 单 y 轴折线图
```js
/** 
* 单 y 轴折线图
* @param    {string}  chartId      图表ID
* @param    {array}   x            x 轴列表
* @param    {array}   yList        y 轴数据列表，列表里的每个元素都是一组数据
* @param    {array}   yNameList    各组数据的名称，与 yList 对应
* @param    {string}  yAxisName    y 坐标轴的名称
*/
function drawLineChart(chartId, x, yList, yNameList, yAxisName) {
  if (document.getElementById(chartId) !== null) {  // 如果实例不为空，销毁实例
    echarts.dispose(document.getElementById(chartId))
  }
  // 初始化 echarts 实例
  var myChart = echarts.init(document.getElementById(chartId))
  // 指定图表的配置项和数据
  var option = {
    grid: {
      top: '15%',
      bottom: '15%',
      left: '5%',
      right: '5%',
      containLabel: true,
    },
    toolbox: {
      feature: {
        saveAsImage: {}
      },
      showTitle: false,
    },
    legend: {
      data: yNameList,
      top: 0,
      left: '7%',
      icon: 'rect',
      itemWidth: 12,
      itemHeight: 12,
      textStyle: {
        color: '#999',
        fontSize: 12
      },
    },
    tooltip: {
      trigger: 'axis',
      axisPointer: {
        type: 'cross',
      }
    },
    xAxis: {
      type: 'category',
      boundaryGap: false,  //折线图中用于坐标轴刻度与标签对齐
      data: x,
      axisLabel: {
        textStyle: {
          fontSize: 12,
          color: '#999'
        },
      },
      axisLine: {lineStyle: {color: '#999'}},
    },
    yAxis: {
      type: 'value',
      name: yAxisName,
      nameTextStyle: {
        color: '#fff',
        fontSize: 9,
        padding: [0, 0, -5, 0]
      },
      axisLabel: {
        textStyle: {
          fontSize: 12,
          color: '#999'
        },
      },
      axisLine: {show: false},
      axisTick: {show: false},
      splitLine: {
        lineStyle: {
          type: 'dashed',
          color: '#666'
        }
      },
    },
    series: [],
  }
  var colorList = [
    ['rgba(175, 76, 250, 1)', 'rgba(175, 76, 250, 0.5)', 'rgba(175, 76, 250, 0)'],
    ['rgba(0, 150, 136, 1)', 'rgba(0, 150, 136, 0.5)', 'rgba(0, 150, 136, 0)'],
    ['rgba(0, 145, 255, 1)', 'rgba(0, 145, 255, 0.5)', 'rgba(0, 145, 255, 0)'],
  ]  //colorList 的长度最好不小于 yList 的长度
  for (var i = 0; i < yList.length; i++) {
    var dataItem = {
      data: yList[i],
      type: 'line',
      name: yNameList[i],
      // symbol: function (value, params) {return 'triangle'},
      showSymbol: false,
      // showAllSymbol: true,
      smooth: true,
      itemStyle: {},
      areaStyle: {},
    }
    if (i < colorList.length) {
      dataItem.itemStyle = {color: colorList[i][0]}
      dataItem.areaStyle = {
        color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
          {offset: 0, color: colorList[i][1]},
          {offset: 1, color: colorList[i][2]}
        ])
      }
    }
    option.series.push(dataItem)
  }
  // 使用刚指定的配置项和数据显示图表
  myChart.setOption(option)
  window.addEventListener('resize', function () {  // 在容器大小发生改变时重绘图表
    myChart.resize()
  })
}
```


#### 单 y 轴柱状图
```js
/** 
* 单 y 轴柱状图
* @param    {string}   chartId      图表ID
* @param    {array}    x            x 轴列表
* @param    {array}    yList        y 轴数据列表，列表里的每个元素都是一组数据
* @param    {array}    yNameList    各组数据的名称，与 yList 对应
* @param    {string}   yAxisName    y 坐标轴的名称
* @param    {array}    colorList    调色盘颜色列表
* @param    {boolean}  average      是否显示平均值
*/
function drawBarChart(chartId, x, yList, yNameList, yAxisName, colorList=[], average=false) {
  if (document.getElementById(chartId) !== null) {  // 如果实例不为空，销毁实例
    echarts.dispose(document.getElementById(chartId))
  }
  // 初始化 echarts 实例
  var myChart = echarts.init(document.getElementById(chartId))
  // 指定图表的配置项和数据
  var option = {
    grid: {
      top: '15%',
      bottom: '15%',
      left: '5%',
      right: '5%',
      containLabel: true,
    },
    toolbox: {
      feature: {
        saveAsImage: {}
      },
      showTitle: false,
    },
    legend: {
      data: yNameList,
      top: 0,
      left: '7%',
      icon: 'rect',
      itemWidth: 12,
      itemHeight: 12,
      textStyle: {
        color: '#999',
        fontSize: 12
      },
    },
    tooltip: {
      trigger: 'axis',
      axisPointer: {
        type: 'shadow',
      }
    },
    xAxis: {
      type: 'category',
      data: x,
      axisLabel: {
        textStyle: {
          fontSize: 12,
          color: '#999'
        },
      },
      axisLine: {lineStyle: {color: '#999'}},
      //axisTick: {alignWithLabel: true},  //坐标轴刻度与标签对齐
    },
    yAxis: {
      type: 'value',
      name: yAxisName,
      nameTextStyle: {
        color: '#fff',
        fontSize: 9,
        padding: [0, 0, -5, 0]
      },
      axisLabel: {
        textStyle: {
          fontSize: 12,
          color: '#999'
        },
      },
      axisLine: {
        lineStyle: {
          color: '#666'
        }
      },
      // axisTick: {show: false},
      splitLine: {
        lineStyle: {
          type: 'dashed',
          color: '#666',
        }
      },
    },
    series: [],
  }
  if (colorList.length > 0) {
    option.color = colorList  
  }
  for (var i = 0; i < yList.length; i++) {
    var dataItem = {
      data: yList[i],
      type: 'bar',
      name: yNameList[i],
      barWidth: 10,
      barGap: 0,  //不同系列的柱间距离
      itemStyle: {},
      markLine: {},
    }
    if (average) {
      dataItem.markLine = {
        data: [
          {type: 'average', name: '平均值'}
        ],
        symbol: 'none',
      }
    }
    option.series.push(dataItem)
  }
  // 使用刚指定的配置项和数据显示图表
  myChart.setOption(option)
  window.addEventListener('resize', function () {  // 在容器大小发生改变时重绘图表
    myChart.resize()
  })
}
```


#### 双 y 轴折柱混合图
```js
/** 
* 双 y 轴折柱混合图
* @param    {string}  chartId          图表ID
* @param    {array}   x                x 轴列表
* @param    {array}   yBarList         y 轴柱图数据列表，列表里的每个元素都是一组柱图数据
* @param    {array}   yBarNameList     各组柱图数据的名称，与 yBarList 对应
* @param    {array}   yLineList        y 轴折线数据列表，列表里的每个元素都是一条折线数据
* @param    {array}   yLineNameList    各条折线数据的名称，与 yLineList 对应
* @param    {string}  yAxisList        y 坐标轴的名称列表
*/
function drawLineBarChart(chartId, x, yBarList, yBarNameList, yLineList, yLineNameList, yAxisList) {
  if (document.getElementById(chartId) !== null) {  // 如果实例不为空，销毁实例
    echarts.dispose(document.getElementById(chartId))
  }
  var max1 = Math.max.apply(null, yBarList.join(",").split(","))
  var max2 = Math.max.apply(null, yLineList.join(",").split(","))
  // 初始化 echarts 实例
  var myChart = echarts.init(document.getElementById(chartId))
  // 指定图表的配置项和数据
  var option = {
    grid: {
      top: '10%',
      bottom: '5%',
      left: '5%',
      right: '5%',
      containLabel: true,
    },
    tooltip: {
      trigger: 'axis',
      axisPointer: {
        type: 'cross',
        label: {
          backgroundColor: '#333',
        },
        crossStyle: {
          color: '#999',
        },
      }
    },
    legend: {
      data: yBarNameList.concat(yLineNameList),
      textStyle: {
        color: '#333'
      }
    },
    xAxis: {
      data: x,
      axisLine: {
        lineStyle: {
          color: '#999'
        }
      },
      axisLabel: {
        textStyle: {
          fontSize: 9,
          color: '#999'
        },
      },
      axisPointer: {
        type: 'shadow'
      }
    },
    yAxis: [{
      type: 'value',
      position: 'left',
      name: yAxisList[0],
      nameTextStyle: {
        color: '#000',
        fontSize: 12,
        padding: [0, 0, -4, 0]
      },
      min: 0,
      max: Math.ceil(max1 / 9.5) * 10,
      interval: Math.ceil(max1 / 9.5) * 2,
      axisLabel: {
        textStyle: {
          fontSize: 12,
          color: '#000'
        },
      },
      axisLine: {
        lineStyle: {
          color: '#999'
        }
      },
      axisTick: {show: false},
      splitLine: {
        show: true,
        lineStyle: {
          color: '#333'
        }
      },
    }, {
      type: 'value',
      position: 'right',
      name: yAxisList[1],
      nameTextStyle: {
        color: '#000',
        fontSize: 12,
        padding: [0, 0, -4, 0]
      },
      min: 0,
      max: Math.ceil(max2 / 9.5) * 10,
      interval: Math.ceil(max2 / 9.5) * 2,
      axisLabel: {
        textStyle: {
          fontSize: 12,
          color: '#000'
        },
      },
      axisLine: {
        lineStyle: {
          color: '#999'
        }
      },
      axisTick: {show: false},
      splitLine: {
        show: false,
      },
    }],
    series: [],
  }
  for (var i = 0; i < yLineList.length; i++) {
    var lineColorList = ['#369ef2', '#36BEF6']
    var lineItem = {
      data: yLineList[i],
      name: yLineNameList[i],
      type: 'line',
      symbol: 'emptyCircle',
      // showAllSymbol: true,
      smooth: true,
      symbolSize: 10,
      itemStyle: {},
    }
    if (i < lineColorList.length) {
      lineItem.itemStyle = {color: lineColorList[i]}
    }
    option.series.push(lineItem)
  }
  for (var j = 0; j < yBarList.length; j++) {
    var barColorList = ['#369ef2', '#36BEF6']
    var barItem = {
      yAxisIndex: 1,
      data: yBarList[j],
      name: yBarNameList[j],
      type: 'bar',
      barWidth: 10,
      barGap: 0,
      itemStyle: {},
    }
    if (j < barColorList.length) {
      barItem.itemStyle = {color: barColorList[j]}
    }
    option.series.push(barItem)
  }
  // 使用刚指定的配置项和数据显示图表
  myChart.setOption(option)
  window.addEventListener('resize', function () {  // 在容器大小发生改变时重绘图表
    myChart.resize()
  })
}
```


#### 饼图
```js
/** 
* 饼图
* @param    {string}  chartId      图表ID
* @param    {array}   valueList    各数据项的值
* @param    {array}   nameList     各数据项的名称，与 valueList 对应
* @param    {string}  seriesName   系列名称
* @param    {array}   colorList    调色盘颜色列表
*/
function drawPieChart(chartId, nameList, valueList, seriesName=null, colorList=[]) {
  if (document.getElementById(chartId) !== null) {  // 如果实例不为空，销毁实例
    echarts.dispose(document.getElementById(chartId))
  }
  // 初始化 echarts 实例
  var myChart = echarts.init(document.getElementById(chartId))
  var seriesData = []
  for (var i = 0; i < valueList.length; i++) {
    seriesData.push({
      name: nameList[i],
      value: valueList[i],
    })
  }
  // 指定图表的配置项和数据
  var option = {
    tooltip: {
      trigger: 'item',
      //formatter: '{a}<br/>{b}: {c} ({d}%)'
    },
    legend: {
      orient: 'vertical',
      top: 'middle',
      left: 0,
      data: nameList,
    },
    series: [
      {
        type: 'pie',
        radius: '50%',
        center: ['50%', '50%'],
        label: {
          alignTo: 'edge',  //标签对齐，需 echarts5 支持
        },
        data: seriesData,
      }
    ]
  }
  if (seriesName) {
    option.series[0].name = seriesName
  }
  if (colorList) {
    option.color = colorList
  }
  // 使用刚指定的配置项和数据显示图表
  myChart.setOption(option)
  window.addEventListener('resize', function () {  // 在容器大小发生改变时重绘图表
    myChart.resize()
  })
}
```


#### 圆环图
```js
/** 
* 圆环图
* @param    {string}  chartId      图表ID
* @param    {array}   valueList    各数据项的值

    echarts.dispose(document.getElementById(chartId))
```
